import java.util.Scanner;
class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner (System.in);
        int num = sc.nextInt();
        if(num>100||num<=0){
            System.out.println("ENTER VALID NUMBER");
        }
        else{
            int multiply=num;
            while(multiply<100){
                System.out.println(multiply);
                multiply+=num;
                
            }
        }
       
        
    }
}
