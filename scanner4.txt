import java.util.Scanner;
class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner (System.in);
        int num = sc.nextInt();
        int greattest=1;
        if(num<=0){
            System.out.println("write valid input");
        }
        int i=1;
        while(i<num){
            if(num%i==0){
               greattest=i;
            }
            i++;
        }
        System.out.println(greattest);
    }
}
