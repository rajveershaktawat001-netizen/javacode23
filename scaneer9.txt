import java.util.Scanner;
public class Main{
    public static void main(String[]args){
        Scanner sc = new Scanner(System.in);
       int num = sc.nextInt();
       int original = num;
       int sum=0;
       while(original!=0){
           int digit = original%10;
           sum+=digit;
           original/=10;
       }
       System.out.print(sum);
        
    }
}
